' ── Observer Pattern ─────────────────────────────────────────
package "Observer Pattern" {

  abstract class Subject {
    - observers : List<Observer>
    + attach(observer : Observer) : void
    + detach(observer : Observer) : void
    + notify(event : String) : void
  }

  interface Observer {
    + update(vehicle : Vehicle, event : String) : void
  }

  class UserNotifier {
    + update(vehicle : Vehicle, event : String) : void
  }

  class AdminDashboard {
    + update(vehicle : Vehicle, event : String) : void
  }

  class RecommendationService {
    + update(vehicle : Vehicle, event : String) : void
  }

  Observer <|.. UserNotifier
  Observer <|.. AdminDashboard
  Observer <|.. RecommendationService
  Subject "1" --> "0..*" Observer : notifies
}

' ── Vehicle + State + Factory ─────────────────────────────────
package "Vehicle Management" {

  abstract class Vehicle {
    + kind : String
    + status : String
    + dailyRate : Decimal
    + city : String
    + reserve() : void
    + returnVehicle() : void
    + sendToMaintenance() : void
    + completeMaintenance() : void
  }

  class Car {
    + fuelType : String
    + bodyStyle : String
  }

  class Bike {
    + bikeType : String
    + hasMotor : Boolean
  }

  class Scooter {
    + engineCC : Integer
    + isElectric : Boolean
  }

  Vehicle <|-- Car
  Vehicle <|-- Bike
  Vehicle <|-- Scooter
  Vehicle --|> Subject

  ' State Pattern
  abstract class VehicleState {
    + reserve(v : Vehicle) : void
    + confirm(v : Vehicle) : void
    + returnVehicle(v : Vehicle) : void
    + sendToMaintenance(v : Vehicle) : void
    + completeMaintenance(v : Vehicle) : void
  }

  class AvailableState {
    + reserve(v : Vehicle) : void
    + sendToMaintenance(v : Vehicle) : void
  }

  class ReservedState {
    + confirm(v : Vehicle) : void
  }

  class InUseState {
    + returnVehicle(v : Vehicle) : void
  }

  class MaintenanceState {
    + completeMaintenance(v : Vehicle) : void
  }

  VehicleState <|-- AvailableState
  VehicleState <|-- ReservedState
  VehicleState <|-- InUseState
  VehicleState <|-- MaintenanceState
  Vehicle --> VehicleState : delegates to >

  ' Factory Pattern
  abstract class VehicleFactory {
    + createVehicle() : Vehicle
  }

  class CarFactory {
    + createVehicle() : Vehicle
  }

  class BikeFactory {
    + createVehicle() : Vehicle
  }

  class ScooterFactory {
    + createVehicle() : Vehicle
  }

  VehicleFactory <|-- CarFactory
  VehicleFactory <|-- BikeFactory
  VehicleFactory <|-- ScooterFactory
  CarFactory     ..> Car     : creates
  BikeFactory    ..> Bike    : creates
  ScooterFactory ..> Scooter : creates
}

package "Public Transport (Adapter + Facade)" {

  interface TransitProvider {
    + getNearbyStops(lat : Float, lon : Float, radius : Integer) : List
    + getNextDepartures(stopId : String) : List
  }

  class GTFSAdapter {
    + getNearbyStops(lat : Float, lon : Float, radius : Integer) : List
    + getNextDepartures(stopId : String) : List
  }

  class CityAPIAdapter {
    + getNearbyStops(lat : Float, lon : Float, radius : Integer) : List
    + getNextDepartures(stopId : String) : List
  }

  class TransitFacade {
    - bus : GTFSAdapter
    - metro : CityAPIAdapter
    + getNearbyStops(lat : Float, lon : Float, radius : Integer) : List
    + getNextDepartures(stopId : String) : List
  }

  class TransitStop {
    + name : String
    + location : String
  }

  class TransitSchedule {
    + route : String
    + destination : String
    + departureTime : Time
  }

  TransitProvider <|.. GTFSAdapter
  TransitProvider <|.. CityAPIAdapter
  TransitFacade o-- GTFSAdapter    : bus
  TransitFacade o-- CityAPIAdapter : metro
  TransitFacade "1" --> "1..*" TransitStop : serves
  TransitStop "1" --> "0..*" TransitSchedule : schedules
}
