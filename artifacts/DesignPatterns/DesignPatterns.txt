  + reserve(v : Vehicle) : void
  + confirm(v : Vehicle) : void
  + returnVehicle(v : Vehicle) : void
  + sendToMaintenance(v : Vehicle) : void
  + completeMaintenance(v : Vehicle) : void
}

class AvailableState {
  + reserve(v : Vehicle) : void
  + sendToMaintenance(v : Vehicle) : void
}

  + confirm(v : Vehicle) : void
}

class InUseState {
  + returnVehicle(v : Vehicle) : void
}

class MaintenanceState {
  + completeMaintenance(v : Vehicle) : void
}

  InvalidTransitionError.
