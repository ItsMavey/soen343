' ────────────────────────────────────────────────────────────
'  3. STRATEGY — Itinerary Planning
'     (booking/trip_strategies.py)
' ────────────────────────────────────────────────────────────
@startuml
title Strategy Pattern — Multimodal Itinerary Planning

abstract class ItineraryStrategy {
  + {abstract} plan(slat: float, slng: float,\n  elat: float, elng: float): dict
  # _leg(mode, label, ...): dict
}

class VehicleOnlyStrategy {
  + plan(...): dict
  ' walk → drive → park → walk
}

class TransitFirstStrategy {
  + plan(...): dict
  ' walk → transit → walk → drive → park → walk
}

class "trip_plan()\n«view»" as TripView {
  + run strategies in priority order
  + return all feasible options
}

note right of TransitFirstStrategy
  Falls back to None if no transit
  stops found near origin/destination.
  Caller skips None results.
end note

ItineraryStrategy <|-- VehicleOnlyStrategy
ItineraryStrategy <|-- TransitFirstStrategy
TripView ..> ItineraryStrategy : selects and runs
@enduml

' ────────────────────────────────────────────────────────────
'  5. OBSERVER — Notifications  (booking/observers.py)
' ────────────────────────────────────────────────────────────
@startuml
title Observer Pattern — Notifications

abstract class Subject {
  - _observers : list
  + attach(observer: Observer): void
  + detach(observer: Observer): void
  + _notify_observers(event: str): void
}

interface Observer {
  + {abstract} update(vehicle: Vehicle, event: str): void
}

class UserNotifier {
  + update(vehicle, event): void
  ' fires on: AVAILABLE
}

class AdminDashboard {
  + update(vehicle, event): void
  ' fires on: MAINTENANCE
}

class RecommendationService {
  + update(vehicle, event): void
  ' fires on: RETURNED
}

class Vehicle {
  + send_to_maintenance(): void
  + complete_maintenance(): void
  + return_vehicle(): void
}

class Notification {
  + user : User
  + event_type : String
  + message : String
  + is_read : Boolean
}

note right of Vehicle
  Vehicle inherits Subject.
  Each observer creates one
  Notification record per event.
end note

Observer <|.. UserNotifier
Observer <|.. AdminDashboard
Observer <|.. RecommendationService
Vehicle --|> Subject
Subject --> Observer : notifies
UserNotifier ..> Notification : creates
AdminDashboard ..> Notification : creates
RecommendationService ..> Notification : creates

  Clients (views, trip planner)
  depend only on TransitFacade.

  transparent to all callers.
